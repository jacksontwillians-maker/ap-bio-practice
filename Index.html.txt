// Practice variables
let pi=0,sel=null;
function startPractice(){hideAll();document.getElementById("practice").classList.remove("hidden");pi=0;loadP();}
function loadP(){
let unit=units[document.getElementById("unitSelect").value];
const q=unit.mcq[pi];
document.getElementById("pQ").textContent=q.q;
let container=document.getElementById("pChoices");
container.innerHTML="";
document.getElementById("pFeedback").textContent="";
q.c.forEach((c,i)=>{
  let d=document.createElement("div");
  d.className="choice"; d.textContent=c;
  d.onclick=()=>{document.querySelectorAll("#pChoices .choice").forEach(x=>x.classList.remove("selected"));d.classList.add("selected"); sel=i;}
  container.appendChild(d);
});
}
function checkPractice(){
let unit=units[document.getElementById("unitSelect").value];
const q=unit.mcq[pi];
document.querySelectorAll("#pChoices .choice").forEach((d,i)=>{
  if(i===q.a)d.classList.add("correct");
  if(i===sel && i!==q.a)d.classList.add("wrong");
});
document.getElementById("pFeedback").textContent = sel===q.a ? "Correct!" : "Incorrect: "+q.e;
}
function nextPractice(){pi++; let unit=units[document.getElementById("unitSelect").value); if(pi<unit.mcq.length) loadP(); else {alert("You finished all questions for this unit!"); pi=0; loadP();}}

// FRQ
let fi=0;
function startFRQ(){hideAll();document.getElementById("frq").classList.remove("hidden");fi=0;loadFRQ();}
function loadFRQ(){let unit=units[document.getElementById("unitSelect").value];document.getElementById("frqPrompt").textContent=unit.frq[fi].p;document.getElementById("frqSample").textContent="";}
function showFRQ(){let unit=units[document.getElementById("unitSelect").value];document.getElementById("frqSample").textContent=unit.frq[fi].s; fi=(fi+1)%unit.frq.length;}

// Matching
let pick=null;
function startMatching(){
hideAll();document.getElementById("matching").classList.remove("hidden");
let unit=units[document.getElementById("unitSelect").value];
let termsDiv=document.getElementById("terms"); let defsDiv=document.getElementById("defs");
termsDiv.innerHTML=""; defsDiv.innerHTML=""; document.getElementById("matchFeedback").textContent="";
unit.match.forEach(pair=>{
  let t=document.createElement("div"); t.className="match"; t.textContent=pair[0]; t.onclick=()=>pick=pair[1]; termsDiv.appendChild(t);
  let d=document.createElement("div"); d.className="match"; d.textContent=pair[1]; d.onclick=()=>{document.getElementById("matchFeedback").textContent=d.textContent===pick?"Correct!":"Try again";}; defsDiv.appendChild(d);
});
}
